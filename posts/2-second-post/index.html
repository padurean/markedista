  protected val metrics = Metrics.openTSDBMetrics

    val metric = "assetEngine.leadershipState"
    val dp = TSDataPoint(
      metric = metric,
      timestamp = now.getMillis,
      value = if (status.isMaster) 1 else 0,
      tags = Map("host" -> s"${status.hostname}/${status.address}")
